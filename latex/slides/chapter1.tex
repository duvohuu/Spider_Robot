\section{Nhận Biết Nhu Cầu}

% ---- Slide 1.1: Bối cảnh + Nhu cầu ----
\begin{frame}{Bối Cảnh Và Nhu Cầu Thực Tiễn}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{Robot di động truyền thống:}
      \begin{itemize}
        \item Bánh xe / dải xích -- hạn chế trên địa hình gồ ghề
        \item Kém linh hoạt trong không gian chật hẹp
      \end{itemize}
      \vspace{0.2cm}
      \textbf{Nhu cầu thực tiễn cần đáp ứng:}
      \begin{itemize}
        \item \textbf{Đa địa hình:} Mỗi chân điều chỉnh độc lập
        \item \textbf{Ổn định:} Tripod gait -- luôn $\geq 3$ chân tiếp đất
        \item \textbf{Realtime:} IK + đồng bộ 18 servo
        \item \textbf{Chi phí:} Servo MG996R, ESP32, in 3D
        \item \textbf{Giáo dục:} Nền tảng nghiên cứu cơ điện tử
      \end{itemize}
    \column{0.45\textwidth}
      \centering
      \includegraphics[width=\linewidth, height=0.62\textheight, keepaspectratio]{robot.png}
      \\\small\textit{Robot nhện 6 chân của nhóm}
  \end{columns}
\end{frame}
