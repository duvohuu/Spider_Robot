  \end{block}

    \column{0.48\textwidth}
      \textbf{Ưu điểm nổi bật:}
      \begin{itemize}
        \item Thăng bằng và thích nghi địa hình vượt trội
        \item Bước qua chướng ngại vật, leo dốc
        \item Duy trì ổn định khi một số chân mất chức năng
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{Thách thức tích hợp:}
      \begin{itemize}
        \item Cơ khí (động học, phân tích lực)
        \item Điện tử (driver servo, mạch điều khiển)
        \item Lập trình nhúng (IK, gait controller)
        \item Truyền thông không dây

% ---- Slide 1.3: Nhu cầu thực tiễn ----
\begin{frame}{Nhu Cầu Thực Tiễn Cần Đáp Ứng}
  \begin{itemize}
    \item<1-> \textbf{Di chuyển đa địa hình:} Hệ thống chân nhiều bậc tự do,
          mỗi chân điều chỉnh độc lập theo bề mặt tiếp xúc
    \item<2-> \textbf{Tính ổn định:} Cấu hình 6 chân (hexapod) -- thuật toán
          dáng đi tam giác (tripod gait), luôn có $\geq 3$ chân tiếp đất
    \item<3-> \textbf{Điều khiển thời gian thực:} Tính toán động học nghịch
          (Inverse Kinematics) và đồng bộ toàn bộ khớp servo
    \item<4-> \textbf{Chi phí hợp lý:} Servo MG996R, vi điều khiển Arduino/STM32,
          kết cấu in 3D / cắt laser
    \item<5-> \textbf{Nền tảng nghiên cứu \& giáo dục:} Tích hợp kiến thức cơ
          học, điện tử, lập trình nhúng và tự động hóa
  \end{itemize}
\end{frame}
