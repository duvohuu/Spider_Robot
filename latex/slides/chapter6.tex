% ---- Slide 6.2: Tính toán lựa chọn servo ----
\begin{frame}{Tính Toán Lựa Chọn Servo}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \centering
      \includegraphics[width=0.9\linewidth, height=0.52\textheight, keepaspectratio]{load_model.png}
      \\\small\textit{Mô hình tải trọng -- 3 chân tiếp đất}
    \column{0.50\textwidth}
      \small
      \textbf{Phản lực (3 chân đỡ):}
      $N_3 = W/3 = 19.62/3 \approx 6.54\;\text{N}$

      \vspace{0.15cm}
      \textbf{Mômen khớp Femur (lớn nhất):}
      $T_h = N_3 (L_1{+}L_2{+}L_3) \approx 23.2\;\text{kgf.cm}$

      \vspace{0.15cm}
      \textbf{Mômen yêu cầu ($K{=}1.3$):}
      $T_\text{cần} = 1.3 \times 23.2 \approx 30.2\;\text{kgf.cm}$

      \vspace{0.15cm}
      \begin{alertblock}{Chọn Servo DS3218}
        \small Moment max $\approx 20$--25 kgf.cm tại 6V\\
        Kết hợp bố trí cơ khí tối ưu
      \end{alertblock}
  \end{columns}
\end{frame}
