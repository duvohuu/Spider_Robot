% ---- Slide 4.1: MATLAB - Hệ tọa độ ----
\begin{frame}{Mô Phỏng MATLAB -- Hệ Tọa Độ}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{Ba hệ tọa độ sử dụng:}
      \begin{itemize}
        \item \textbf{Hệ body:} gắn với thân robot -- khai báo vị trí khớp hông
        \item \textbf{Hệ world:} biểu diễn robot trong không gian 3D toàn cục
        \item \textbf{Hệ local:} gắn với từng chân -- giải IK bằng công thức chung
      \end{itemize}
      \vspace{0.2cm}
      \begin{block}{Lợi ích}
        Chuẩn hóa 6 chân về cùng hệ quy chiếu $\Rightarrow$ dùng chung 1 hàm IK/FK
      \end{block}
    \column{0.45\textwidth}
      \centering
      \includegraphics[width=\linewidth]{coordinate_system.png}
      \\\small\textit{Sơ đồ hệ tọa độ robot}
  \end{columns}
\end{frame}

% ---- Slide 4.2: MATLAB - Quy trình mô phỏng ----
\begin{frame}{Mô Phỏng MATLAB -- Quy Trình}
  \begin{enumerate}
    \item Xác định hình học: vị trí hip, chiều dài Coxa--Femur--Tibia
    \item Thiết lập dáng đi: chu kỳ $T$, hệ số $\beta$, chiều dài bước, độ nhấc chân
    \item Sinh quỹ đạo bàn chân (stance / swing) theo pha trong hệ body
    \item Cộng vị trí thân $\Rightarrow$ thu tọa độ hip và bàn chân trong hệ world
    \item Đổi về hệ local từng chân (ma trận quay $\psi_i$)
    \item \textbf{Giải IK} $\Rightarrow$ $(q_1, q_2, q_3)$ cho từng chân
    \item \textbf{Giải FK} $\Rightarrow$ tái dựng $P_0 \to P_1 \to P_2 \to P_3$ vẽ robot
    \item Dựng support polygon -- kiểm tra ổn định hình học
  \end{enumerate}
\end{frame}
