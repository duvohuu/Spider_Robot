        \item \textbf{Femur:} Nâng/hạ chân -- điều chỉnh chiều cao thân

      \begin{itemize}
        \item Đối xứng qua trục dọc thân
        \item Mạch + pin ở trung tâm (giảm lệch trọng tâm)
        \item Module chân tháo lắp độc lập
      \end{itemize}

% ---- Slide 5.3: Lựa chọn linh kiện điện ----
\begin{frame}{Thiết Kế Khái Quát Điện -- Lựa Chọn Linh Kiện}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \begin{block}{Vi điều khiển: \textbf{ESP32}}
        \small 240 MHz dual-core, RAM 520 KB\\
        WiFi + BT -- IK realtime 6 chân
      \end{block}
      \vspace{0.1cm}
      \begin{block}{Động cơ: \textbf{Servo MG996R $\times$ 18}}
        \small Bánh răng kim loại\\
        Moment 9.4--11 kgf.cm (4.8--6V)
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Mở rộng PWM: \textbf{PCA9685 $\times$ 2}}
        \small 32 kênh PWM qua I$^2$C\\
        Giải quyết giới hạn kênh vi điều khiển
      \end{block}
      \vspace{0.1cm}
      \begin{block}{Nguồn: \textbf{Hai nguồn tách biệt}}
        \small Servo: LiPo 3S 11.1V\\
        Điều khiển: cell 7.4V 18650\\
        Cách ly nhiễu -- tránh reset ESP32
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Thiết Kế Khái Quát Phần Mềm}
  \textbf{Kiến trúc phân tầng 4 lớp:}
  \vspace{0.2cm}
  \begin{enumerate}
    \item \textbf{Communication Layer:} Web App $\xrightarrow{\text{WebSocket/JSON}}$ ESP32 WiFi SoftAP
    \item \textbf{Coordination Layer:} FreeRTOS -- quản lý trạng thái Idle $\leftrightarrow$ Running,\\
          tích hợp cảm biến SRF05 (tránh vật cản)
    \item \textbf{Motion Layer:} Gait controller tính tọa độ đầu chân $(x,y,z)$\\
          $\Rightarrow$ IK $\Rightarrow$ góc khớp $(\theta_1, \theta_2, \theta_3)$
    \item \textbf{Actuation Layer:} PCA9685 chuyển góc $\to$ PWM $\to$ 18 servo
  \end{enumerate}
  \vspace{0.2cm}
  \begin{block}{Giao diện người dùng}
    \textbf{React + Vite} SPA -- WebGL 3D viewer -- không cần cài đặt\\
    Truy cập qua trình duyệt bất kỳ kết nối WiFi SoftAP của robot
  \end{block}
\end{frame}
