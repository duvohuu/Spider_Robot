% ---- Slide 3.2: Động học thuận ----
\begin{frame}{Động Học Thuận (Forward Kinematics)}
  \textbf{Cho góc khớp $(q_1, q_2, q_3)$ $\Rightarrow$ tìm vị trí đầu chân $(X, Y, Z)$:}
  \vspace{0.2cm}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \begin{align}
        p &= L_1 + L_2\cos q_2 + L_3\cos(q_2+q_3) \nonumber\\
        X &= p\cos q_1 \nonumber\\
        Y &= p\sin q_1 \nonumber\\
        Z &= L_2\sin q_2 + L_3\sin(q_2+q_3) \nonumber
      \end{align}
    \column{0.46\textwidth}
      \centering
      \includegraphics[width=\linewidth]{leg_3dof.png}
      \\\small\textit{Mô hình 3-DOF một chân}
  \end{columns}
\end{frame}

% ---- Slide 3.3: Động học nghịch ----
\begin{frame}{Động Học Nghịch (Inverse Kinematics)}
  \textbf{Cho vị trí đầu chân $(X, Y, Z)$ $\Rightarrow$ tìm góc khớp $(q_1, q_2, q_3)$:}
  \vspace{0.1cm}
  \begin{columns}[T]
    \column{0.48\textwidth}
      \textbf{Bước 1} -- Góc Coxa:
      $q_1 = \mathrm{atan2}(Y,\, X)$

      \vspace{0.2cm}
      \textbf{Bước 2} -- Tầm với:
      $d = \sqrt{X^2+Y^2} - L_1$,\quad $D = \sqrt{d^2+Z^2}$

      \vspace{0.2cm}
      \textbf{Bước 3} -- Góc Tibia (định lý cosin):
      $\cos q_3 = \dfrac{D^2 - L_2^2 - L_3^2}{2L_2 L_3}$

      \vspace{0.2cm}
      \textbf{Bước 4} -- Góc Femur:
      $q_2 = \mathrm{atan2}(Z,d) - \mathrm{atan2}(L_3\sin q_3,\, L_2+L_3\cos q_3)$
    \column{0.48\textwidth}
      \begin{block}{Lựa chọn cấu hình}
        Elbow-down ($q_3 > 0$) -- đảm bảo khoảng tĩnh không dưới thân robot
      \end{block}
      \vspace{0.3cm}
      \begin{alertblock}{Ứng dụng}
        IK là bài toán cốt lõi: lập trình quỹ đạo bằng tọa độ $(x,y,z)$ Descartes
        rồi tự động tính góc servo
      \end{alertblock}
  \end{columns}
\end{frame}

% ---- Slide 3.4: Thuật toán dáng đi ----
\begin{frame}{Thuật Toán Dáng Đi (Gait Algorithm)}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \centering
      \begin{tabular}{lccc}
        \toprule
        \textbf{Gait} & \textbf{Swing} & \textbf{Duty $\beta$} & \textbf{Ổn định} \\
        \midrule
        Tripod & 3 chân & 0.50 & 3 điểm tựa \\
        Ripple & 2 chân & 0.67 & 4 điểm tựa \\
        Wave   & 1 chân & 0.83 & 5 điểm tựa \\
        \bottomrule
      \end{tabular}
      \vspace{0.3cm}

      \begin{block}{Chọn: Tripod Gait}
        6 chân $\to$ 2 nhóm tam giác (1,3,5) và (2,4,6) xen kẽ nhau
      \end{block}
    \column{0.42\textwidth}
      \textbf{Đặc trưng dáng đi:}
      \begin{itemize}
        \item \textbf{Duty factor $\beta$}: tỉ lệ thời gian chân tiếp đất
        \item \textbf{Swing phase}: chân trên không -- quỹ đạo cung
        \item \textbf{Stance phase}: chân tiếp đất -- đẩy thân tiến
      \end{itemize}
      \vspace{0.2cm}
      \textit{Tripod đảm bảo $\beta > 0.5$ -- luôn có tam giác đỡ bao quanh trọng tâm}
  \end{columns}
\end{frame}
